# v0.6 Snapshot Query and Profile Batch Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build IGESS v0.6 as a deterministic observability layer with exact snapshot queries and YAML-defined profile batch runs.

**Architecture:** Add focused `snapshot.py` and `batch.py` service modules while keeping `cli.py` as orchestration only. Refactor `Simulator` just enough to share one-profile lane execution between existing full scenario runs and exact-time snapshots.

**Tech Stack:** Python 3.12, stdlib `argparse`/`json`/`csv`/`pathlib`, existing IGESS dataclasses, existing `pytest` suite.

Create `src/igess/snapshot.py`:

```python
from __future__ import annotations

import json
from pathlib import Path
from typing import Any

from .policy import PolicyEngine
from .schema import EconomyModel, Event
from .simulator import Simulator


class SnapshotService:
    @classmethod
    def snapshot_at(
        cls,
        model: EconomyModel,
        scenario_id: str,
        profile_id: str,
        time_seconds: int,
    ) -> dict[str, Any]:
        cls._validate(model, scenario_id, profile_id, time_seconds)
        result = Simulator(model).snapshot_at(scenario_id, profile_id, time_seconds)
        state = result.state
        action = PolicyEngine(model).choose_action(profile_id, state)
        return {
            "schema_version": 1,
            "scenario_id": scenario_id,
            "profile_id": profile_id,
            "time_seconds": time_seconds,
            "model_id": model.config.model_id,
            "resources": {
                resource_id: state.resources[resource_id].to_decimal_string()
                for resource_id in sorted(model.resources)
            },
            "generators_owned": {
                key: state.generators_owned[key] for key in sorted(state.generators_owned)
            },
            "upgrades_purchased": sorted(state.upgrades_purchased),
            "unlocked_generators": sorted(state.unlocked_generators),
            "unlocked_upgrades": sorted(state.unlocked_upgrades),
            "milestones_claimed": sorted(state.milestones_claimed),
            "prestige_counts": dict(sorted(state.prestige_counts.items())),
            "total_cps": Simulator(model)._total_cps(profile_id, state).to_decimal_string(),
            "recent_events": [
                event.to_ordered_dict() for event in cls._recent_events(result.events, time_seconds)
            ],
            "next_recommended_action": None
            if action is None
            else {
                "kind": action.kind,
                "item_id": action.item_id,
                "cost_resource": action.cost_resource,
                "cost": action.cost.to_decimal_string(),
                "score": action.score.to_decimal_string(),
            },
        }
```

Use the exact validation messages from the spec.

@classmethod
def write(cls, snapshot: dict[str, Any], output_dir: str | Path) -> None:
    output_dir = Path(output_dir)

    (output_dir / "snapshot.json").write_text(

        newline="\n",

        encoding="utf-8",
        newline="\n",

Create `src/igess/batch.py`:

```python
from __future__ import annotations

import json
from pathlib import Path
from typing import Any

from .builder import ModelBuilder
from .loader import ConfigLoader
from .linter import ConfigLinter
from .outputs import OutputWriter
from .schema import SimulationResult
from .simulator import Simulator


def run_profile_batch(
    config: str | Path,
    tables: str | Path,
    scenario_id: str,
    output_dir: str | Path,
) -> Path:
    raw = ConfigLoader.load(config, tables)
    ConfigLinter.validate(raw)
    model = ModelBuilder.build(raw)
    if scenario_id not in model.scenarios:
        raise ValueError(f"scenario '{scenario_id}' not found")
    scenario = model.scenarios[scenario_id]
    if not scenario.profiles:
        raise ValueError(f"batch scenario '{scenario_id}' has no profiles")
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    lanes = []
    summary: dict[str, Any] = {
        "schema_version": 1,
        "scenario_id": scenario_id,
        "model_id": model.config.model_id,
        "profiles": list(scenario.profiles),
        "by_profile": {},
    }
    for profile_id in scenario.profiles:
        lane = Simulator(model).snapshot_at(
            scenario_id,
            profile_id,
            int(scenario.duration_hours * 3600),
        )
        result = SimulationResult(
            scenario_id=scenario_id,
            timeline=lane.timeline,
            events=lane.events,
        )
        run_dir = output_dir / "profiles" / profile_id
        OutputWriter.write_all(result, run_dir, model)
        lanes.append(
            {
                "profile_id": profile_id,
                "status": "success",
                "run_dir": str(Path("profiles") / profile_id).replace("\\", "/"),
                "artifacts": [
                    "analysis.json",
                    "analysis.md",
                    "events.csv",
                    "events.json",
                    "payback.csv",
                    "run_manifest.json",
                    "timeline.csv",
                    "timeline.json",
                ],
            }
        )
        summary["by_profile"][profile_id] = _summarize_lane(lane)
    manifest = {
        "schema_version": 1,
        "scenario_id": scenario_id,
        "model_id": model.config.model_id,
        "profiles": list(scenario.profiles),
        "lanes": lanes,
    }
    _write_json(output_dir / "batch_manifest.json", manifest)
    _write_json(output_dir / "batch_summary.json", summary)
    return output_dir / "batch_manifest.json"
```

Implement `_summarize_lane` using the last timeline row and lane events.
