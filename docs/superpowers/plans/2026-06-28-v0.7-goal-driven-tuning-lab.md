> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build IGESS v0.7 as a bounded, deterministic tuning recommendation workflow that ranks human-only table changes against designer-defined targets.

**Architecture:** Add tuning config parsing to the existing YAML load/lint pipeline, then add focused tuning services for candidate generation, metric scoring, report writing, CLI orchestration, and dashboard display. **Tech Stack:** Python 3.12, stdlib `argparse`/`json`/`csv`/`dataclasses`/`decimal`/`pathlib`, existing IGESS YAML/Luban export loaders, existing `pytest` suite.

```python
def expand_candidate_values(config: dict[str, Any] | list[Any]) -> list[str]:
    if isinstance(config, list):
        return [str(value) for value in config]
    start = Decimal(str(config["start"]))
    end = Decimal(str(config["end"]))
    step = Decimal(str(config["step"]))
    if step <= 0:
        raise ValueError("tuning range step must be > 0")
    precision = max(
        _decimal_places(str(config["start"])),
        _decimal_places(str(config["end"])),
        _decimal_places(str(config["step"])),
    )
    values = []
    current = start
    while current <= end:
        values.append(f"{current:.{precision}f}")
        current += step
    return values
```

```python
def load_tuning_scenario(raw: RawConfig, scenario_id: str) -> TuningScenario:
    target_config = raw.rules.tuning_targets.get(scenario_id)
    candidate_config = raw.rules.tuning_candidates.get(scenario_id)
    if target_config is None:
        raise ValueError(f"scenario '{scenario_id}' has no tuning_targets")
    if candidate_config is None:
        raise ValueError(f"scenario '{scenario_id}' has no tuning_candidates")
    fields = []
    for item in candidate_config.get("fields", []):
        values = item.get("values")
        if values is None:
            values = expand_candidate_values(item["range"])
        else:
            values = expand_candidate_values(list(values))
        fields.append(
            TuningCandidateField(
                table=str(item["table"]),
                row_id=str(item["row_id"]),
                field=str(item["field"]),
                values=values,
            )
        )
    return TuningScenario(
        scenario_id=scenario_id,
        profile_weights={
            profile: float(weight)
            for profile, weight in sorted(target_config.get("profile_weights", {}).items())
        },
        targets=dict(target_config.get("metrics", {})),
        max_candidates=int(candidate_config.get("max_candidates", 120)),
        candidate_fields=fields,
    )
```

```python
    @classmethod
    def _validate_tuning(cls, raw: RawConfig) -> None:
        for scenario_id in sorted(set(raw.rules.tuning_targets) | set(raw.rules.tuning_candidates)):
            if scenario_id not in raw.rules.scenarios:
                raise ConfigError(f"tuning scenario '{scenario_id}' is not defined")
            scenario = load_tuning_scenario(raw, scenario_id)
            for metric in scenario.targets:
                if metric not in SUPPORTED_TUNING_METRICS:
                    raise ConfigError(f"unsupported tuning target metric '{metric}'")
            total = 1
            for field in scenario.candidate_fields:
                cls._validate_tuning_field(raw, field)
                total *= len(field.values)
            if total > scenario.max_candidates:
                raise ConfigError(
                    f"tuning candidate grid expands to {total} variants, "
                    f"over max_candidates {scenario.max_candidates}"
                )
```

```python
    @classmethod
    def _validate_tuning_field(cls, raw: RawConfig, field) -> None:
        rows = getattr(raw, field.table, None)
        if rows is None:
            raise ConfigError(f"tuning candidate table '{field.table}' not found")
        row = next((item for item in rows if item.id == field.row_id), None)
        if row is None:
            raise ConfigError(f"tuning candidate field {field.key} not found")
        if not hasattr(row, field.field):
            raise ConfigError(f"tuning candidate field {field.key} not found")
        current = getattr(row, field.field)
        try:
            Decimal(str(current))
            for value in field.values:
                Decimal(str(value))
        except (InvalidOperation, ValueError):
            raise ConfigError(f"tuning candidate field {field.key} is not numeric") from None
```

```python
def generate_candidates(scenario: TuningScenario) -> list[TuningCandidate]:
    fields = scenario.candidate_fields
    candidates = []
    products = itertools.product(*(field.values for field in fields))
    for index, values in enumerate(products, start=1):
        changes = [
            TuningChange(
                table=field.table,
                row_id=field.row_id,
                field=field.field,
                value=str(value),
            )
            for field, value in zip(fields, values, strict=True)

        candidates.append(TuningCandidate(candidate_id=f"cand_{index:04d}", changes=changes))
    return candidates
```

```python
from __future__ import annotations

from decimal import Decimal, ROUND_HALF_UP
from typing import Any


def evaluate_targets(
    metrics: dict[str, Any],
    targets: dict[str, Any],
    profile_weights: dict[str, float],
) -> dict[str, Any]:
    results = []
    for metric, config in sorted(targets.items()):
        if metric == "max_purchase_gap_seconds":
            results.extend(_evaluate_profile_max(metric, metrics["purchase_gaps"], config, profile_weights))
        elif metric == "max_unlock_time_seconds":
            results.extend(_evaluate_item_profile_max(metric, metrics["unlock_times"], config, profile_weights))
        elif metric == "max_payback_seconds":
            results.extend(_evaluate_item_profile_max(metric, metrics["payback_seconds"], config, profile_weights))
        elif metric == "min_prestige_gain":
            results.extend(_evaluate_profile_min(metric, metrics["prestige_counts"], config, profile_weights))
        elif metric == "max_prestige_time_seconds":
            results.extend(_evaluate_profile_max(metric, metrics["first_prestige_times"], config, profile_weights))
        else:
            raise ValueError(f"unsupported tuning target metric '{metric}'")
    weighted = sum(Decimal(item["score"]) * Decimal(str(item["weight"])) for item in results)
    total_weight = sum(Decimal(str(item["weight"])) for item in results)
    score = Decimal("0") if total_weight == 0 else weighted / total_weight
    return {
        "passed": all(item["status"] == "passed" for item in results),
        "score": _format_score(score),
        "target_results": results,
    }
```

```python
def _evaluate_profile_max(metric, actuals, targets, weights):
    rows = []
    for profile, target in sorted(targets.items()):
        actual = Decimal(str(actuals.get(profile, 0)))
        expected = Decimal(str(target))
        score = Decimal("1") if actual <= expected else expected / actual
        rows.append(_result(metric, profile, None, actual, expected, score, weights))
    return rows


def _evaluate_profile_min(metric, actuals, targets, weights):
    rows = []
    for profile, target in sorted(targets.items()):
        actual = Decimal(str(actuals.get(profile, 0)))
        expected = Decimal(str(target))
        score = Decimal("1") if actual >= expected else (actual / expected if expected else Decimal("1"))
        rows.append(_result(metric, profile, None, actual, expected, score, weights))
    return rows


def _evaluate_item_profile_max(metric, actuals, targets, weights):
    rows = []
    for item_key, per_profile in sorted(targets.items()):
        for profile, target in sorted(per_profile.items()):
            actual = Decimal(str(actuals.get(profile, {}).get(item_key, 0)))
            expected = Decimal(str(target))
            score = Decimal("1") if actual <= expected else expected / actual
            rows.append(_result(metric, profile, item_key, actual, expected, score, weights))
```

```python
def _result(metric, profile, item_key, actual, expected, score, weights):
    score = max(Decimal("0"), min(Decimal("1"), score))
    return {
        "metric": metric,
        "profile_id": profile,
        "item_key": item_key,
        "actual": _format_score(actual),
        "target": _format_score(expected),
        "score": _format_score(score),
        "weight": str(weights.get(profile, 1.0)),
        "status": "passed" if score == Decimal("1") else "failed",
    }


def _format_score(value: Decimal) -> str:
    rounded = value.quantize(Decimal("0.000001"), rounding=ROUND_HALF_UP)
    if rounded == rounded.to_integral_value():
        return str(rounded.quantize(Decimal("1")))
    return format(rounded.normalize(), "f")
```

```python
def run_tuning(
    config: str | Path,
    tables: str | Path,
    scenario_id: str,
    output_dir: str | Path,
    baseline: str | Path | None = None,
) -> Path:
    raw = ConfigLoader.load(config, tables)
    ConfigLinter.validate(raw)
    scenario = load_tuning_scenario(raw, scenario_id)
    candidates = generate_candidates(scenario)
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)

    ranked = []
    rejected = []
    for candidate in candidates:
        candidate_raw = copy.deepcopy(raw)
        apply_candidate(candidate_raw, candidate)
        try:
            ConfigLinter.validate(candidate_raw)
            model = ModelBuilder.build(candidate_raw)
            result = Simulator(model).run_scenario(scenario_id)
            run_dir = output_dir / "candidates" / candidate.candidate_id
            OutputWriter.write_all(
                result,
                run_dir,
                model,
                overrides=[change.override_label() for change in candidate.changes],
            )
            generate_static_report(run_dir, run_dir / "report")
            metrics = extract_metrics(run_dir)
            score = evaluate_targets(metrics, scenario.targets, scenario.profile_weights)
            ranked.append(_candidate_report(raw, candidate, score, metrics, run_dir, output_dir))
        except Exception as exc:  # noqa: BLE001 - candidate failures are reportable.
            rejected.append(
                {
                    "candidate_id": candidate.candidate_id,
                    "status": "failed",
                    "message": str(exc),
                }
            )

    ranked.sort(key=lambda item: (-float(item["score"]), item["change_magnitude"], item["candidate_id"]))
    for rank, item in enumerate(ranked, start=1):
        item["rank"] = rank
        item.pop("change_magnitude", None)

    report = {
        "schema_version": 1,
        "scenario_id": scenario_id,
        "model_id": raw.rules.model.id,
        "status": "candidates_found" if ranked else "no_candidates",
        "summary": f"{len(ranked)} candidates evaluated successfully.",
        "baseline": None if baseline is None else {"path": str(baseline), "kind": "unclassified"},
        "targets": scenario.targets,
        "candidate_count": len(candidates),
        "best_candidates": ranked[:5],
        "rejected_candidates": rejected,
    }
    return _write_tuning_report(report, output_dir)
```
